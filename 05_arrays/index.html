Man fjerner elementer i et array med funktionen splice()

                    

                        For eksempel dette array: var liste = ["John", "Paul", "Ringo", "George"]
                        liste.splice(0,1)
                        Og nu er listen: ["Paul", "Ringo", "George"]

                    

                
            
                
                    Man kan filtrere et array med funktionen filter()
                    var names = ["Simon", "Silas", "Sune", "Seimi", "Selma", "Seb"]
                    var namesWithA = names.filter( n => {
                        return n.includes('a')
                    } )
                    console.log(namesWithA) // ["Silas", "Selma"]
